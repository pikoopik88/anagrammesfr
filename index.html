let peer, conn; let connections = []; let isHost = false; let fullDictionary = []; let currentRound = { baseWord: "", solutions: [], found: [], timeLeft: 180, active: false }; let timerInterval; const normalize = (str) => str.normalize("NFD").replace(/[\u0300-\u036f]/g, "").toUpperCase(); // 1. INITIALISATION const urlParams = new URLSearchParams(window.location.search); const joinId = urlParams.get('join'); if (joinId) { isHost = false; peer = new Peer(); peer.on('open', (id) => { document.getElementById('status-msg').innerText = "Connexion au Host..."; conn = peer.connect(joinId); setupClientEvents(conn); }); } else { isHost = true; peer = new Peer(); peer.on('open', (id) => { const baseUrl = window.location.href.split('?')[0]; const link = baseUrl + "?join=" + id; document.getElementById('status-msg').innerText = "Tu es le Host !"; document.getElementById('game-link-input').value = link; document.getElementById('share-area').style.display = "block"; document.getElementById('start-btn').style.display = "inline-block"; }); peer.on('connection', (c) => { connections.push(c); // --- NOUVEAU : SYNC SI PARTIE EN COURS --- if (currentRound.active) { c.on('open', () => { c.send({ type: 'SYNC_JOIN', round: currentRound }); }); } c.on('data', (data) => { if (data.type === 'GUESS') handleGuess(data.word); }); }); } async function loadDict() { try { const resp = await fetch('https://raw.githubusercontent.com/words/an-array-of-french-words/master/index.json'); const words = await resp.json(); fullDictionary = words.filter(w => w.length >= 4 && w.length <= 8).map(w => normalize(w)); } catch(e) { console.error("Erreur dico", e); } } // 2. LANCEMENT / REJOUER function setupNewRound() { if (!isHost) return; // Reset local clearInterval(timerInterval); const targetLength = [6, 7, 8][Math.floor(Math.random() * 3)]; const pool = fullDictionary.filter(w => w.length === targetLength); const base = pool[Math.floor(Math.random() * pool.length)]; const baseFreq = {}; for (let l of base) baseFreq[l] = (baseFreq[l] || 0) + 1; const possible = Array.from(new Set(fullDictionary.filter(w => w.length >= 4 && canForm(w, baseFreq)))) .sort((a,b) => a.length - b.length || a.localeCompare(b)); currentRound = { baseWord: base, solutions: possible, found: [], timeLeft: 180, active: true }; // On cache le panneau de fin et on lance chez tout le monde broadcast({ type: 'START', round: currentRound }); startHostTimer(); } function handleGuess(word) { if (!currentRound.active) return; const val = normalize(word); if (currentRound.solutions.includes(val) && !currentRound.found.includes(val)) { currentRound.found.push(val); broadcast({ type: 'UPDATE', word: val, foundCount: currentRound.found.length }); if (currentRound.found.length === currentRound.solutions.length) { broadcast({ type: 'END', title: "PARFAIT !", allSolutions: currentRound.solutions }); } } } function broadcast(data) { connections.forEach(c => { if(c.open) c.send(data); }); handlePeerData(data); // Appliquer à soi-même } function setupClientEvents(c) { c.on('data', (data) => handlePeerData(data)); c.on('open', () => { document.getElementById('status-msg').innerText = "Connecté ! En attente du début..."; }); } function handlePeerData(data) { if (data.type === 'START' || data.type === 'SYNC_JOIN') { currentRound = data.round; currentRound.active = true; document.getElementById('setup-ui').style.display = 'none'; document.getElementById('game-ui').style.display = 'block'; document.getElementById('game-over-panel').style.display = 'none'; document.getElementById('user-input').disabled = false; document.getElementById('user-input').value = ""; renderGrid(); // Afficher les mots déjà trouvés si on rejoint en cours currentRound.found.forEach(w => updateUIFound(w)); // Lancer le chrono visuel startVisualTimer(); } if (data.type === 'UPDATE') { updateUIFound(data.word); currentRound.found.length = data.foundCount; updateScoreText(); } if (data.type === 'END') { showEndScreen(data.title, data.allSolutions); } } // 3. VISUELS ET TIMERS function startHostTimer() { clearInterval(timerInterval); timerInterval = setInterval(() => { currentRound.timeLeft--; if (currentRound.timeLeft <= 0) { broadcast({ type: 'END', title: "TEMPS ÉCOULÉ !", allSolutions: currentRound.solutions }); } }, 1000); } function startVisualTimer() { clearInterval(timerInterval); timerInterval = setInterval(() => { const mins = Math.floor(currentRound.timeLeft / 60); const secs = currentRound.timeLeft % 60; document.getElementById('timer-text').innerText = `${mins}:${secs.toString().padStart(2, '0')}`; document.getElementById('timer-bar').style.width = (currentRound.timeLeft / 180 * 100) + "%"; }, 1000); } function showEndScreen(title, allSolutions) { currentRound.active = false; clearInterval(timerInterval); document.getElementById('user-input').disabled = true; allSolutions.forEach(w => { const el = document.getElementById('word-' + w); if (el && !el.classList.contains('revealed')) { el.innerText = w; el.classList.add('missed'); } }); document.getElementById('result-title').innerText = title; document.getElementById('final-res').innerText = `Score final : ${currentRound.found.length} / ${allSolutions.length}`; document.getElementById('game-over-panel').style.display = 'flex'; if(isHost) { document.getElementById('restart-btn').style.display = 'block'; } } // (Le reste des fonctions utilitaires comme renderGrid, canForm, updateUIFound restent identiques)
